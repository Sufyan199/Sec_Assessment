# Q9: Check password conditions (has upper and lower case letters)
pwd = input("Enter your password: ")  # User input required
upper = any(c.isupper() for c in pwd)
lower = any(c.islower() for c in pwd)
print("Q9 Result:")
print("Yes" if upper and lower else "False")
